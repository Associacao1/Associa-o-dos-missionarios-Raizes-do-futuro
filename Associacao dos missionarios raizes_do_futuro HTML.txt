<script>
function switchLang(lang){
  const langs = ['pt','en'];
  langs.forEach(l=>{
    document.querySelectorAll('[id$="-'+l+'"]').forEach(el=>{
      if(l===lang){ el.classList.remove('hidden'); el.classList.add('fade-in'); }
      else{ el.classList.remove('fade-in'); el.classList.add('hidden'); }
    });
  });
  // Alternar títulos do formulário
  document.getElementById('form-title-pt').classList.toggle('hidden', lang!=='pt');
  document.getElementById('form-title-en').classList.toggle('hidden', lang!=='en');

  // Atualizar placeholders do formulário
  if(lang==='pt'){
    document.getElementById('label-nome').innerText='Seu nome';
    document.getElementById('nome').placeholder='Digite seu nome';
    document.getElementById('label-email').innerText='Email';
    document.getElementById('email').placeholder='Digite seu email';
    document.getElementById('label-mensagem').innerText='Mensagem';
    document.getElementById('mensagem').placeholder='Escreva sua mensagem';
    document.getElementById('btn-submit').innerText='Enviar';
  } else {
    document.getElementById('label-nome').innerText='Your Name';
    document.getElementById('nome').placeholder='Enter your name';
    document.getElementById('label-email').innerText='Email';
    document.getElementById('email').placeholder='Enter your email';
    document.getElementById('label-mensagem').innerText='Message';
    document.getElementById('mensagem').placeholder='Write your message';
    document.getElementById('btn-submit').innerText='Send';
  }
}
</script>
